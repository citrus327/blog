/**
 * polyfill for scrollTo on element
 * @param {HtmlElement} el
 * @param {number} x
 * @param {number} y
 */
function scrollTo(el, x, y) {
  if (!el) {
    return;
  }

  if (el.scrollTo) {
    el.scrollTo(x, y);
  } else {
    el.scrollLeft = x;
    el.scrollTop = y;
  }
}
