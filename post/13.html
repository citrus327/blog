React学习与探索之路（二）- Styled Component

引入方式

  import styled from 'styled-component'

如何定义

普通div

const StyledDiv = styled.div`
  height: 200px;
  width: 200px;
`

组件

import { Button } from '@material-ui/core'
import styled from 'styled-component'

const StyledButton = styled.Button`
  height: 200px;
  width: 200px;
`

覆写子元素

const StyledButton = styled.Button`
  height: 200px;
  width: 200px;

  &.child {
    color: red;
  }
`

传参

const StyledDiv = styled.div`
  background: ${props => props.backgroundColor || 'grey'};
  height: 200px;
  width: 200px;
`

render () {
  return (
      <div>
        <StyledDiv backgroundColor="red">
          <Child/>
        </StyledDiv>
      </div>
    )
}
